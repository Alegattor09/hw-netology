### **Решение 2**
Содержимое gitlab-ci.yml
stages:
  - test
  - build

test:
  stage: test
  image: golang:1.17
  script: 
   - go test .
  tags:
   - netology

build:
  stage: build
  image: docker:latest
  script:
   - docker build .
  tags:
   - netology
